                 function freqDistr(arr) {
                    const count = {};
                    arr.forEach(item => {
                        if (typeof item.valore === 'string' || Number.isInteger(item.valore)){
                            const valore = item.valore
                            if (count[valore] === undefined) {
                                count[valore] = 1;
                            } else {
                                count[valore]++;
                            }
                        }                
                        else {    
                            const interval=parseFloat(item.valore).toFixed(1);
                            const valore = interval.toString() + "-" + ((parseFloat(interval) + 0.1).toFixed(1)).toString() ;
                            if (count[valore] === undefined) {
                                count[valore] = 1;
                            } else {
                                count[valore]++;
                            }
                        }
                        
                    });
                    
                    return count;
                }
        
                function freqDistrJoin(arr1, arr2) {
                    const combinedCounts = {};
                    for (let i = 0; i < arr1.length; i++) {
                        const AgeValue = arr1[i].valore.toString(); 
                        const sportValue = arr2[i].valore;  
        
                        const combinedKey = AgeValue +"-"+ sportValue;
                        
                        if (combinedCounts[combinedKey] === undefined) {
                            combinedCounts[combinedKey] = 1;
                        } else {
                            combinedCounts[combinedKey]++;
                        }
                    }
        
                    return combinedCounts;
                }
