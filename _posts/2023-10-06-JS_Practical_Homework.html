Write a program in JavaScript or TypeScript that creates a window containing a single line, point, circle, rectangle.

Below is the JavaScript code through which you could draw the different shapes 

        
             <script> 
            const canvas = document.getElementById("myCanvas"); // To take the canvas element of the html document
            const canvas1 = document.getElementById("myCanvas1");
            const canvas2 = document.getElementById("myCanvas2");
            const canvas3 = document.getElementById("myCanvas3");
            
            const ctx = canvas.getContext("2d");  //To get the 2D canvas drawing context
            const ctx1 = canvas1.getContext("2d");
            const ctx2 = canvas2.getContext("2d");
            const ctx3 = canvas3.getContext("2d");

            //to draw the circle 
            const centerX = canvas.width / 2;
            const centerY = canvas.height / 2;
            const radius = 50;     // To define the coordinates of the center, of the circle and the radius
            
            ctx.beginPath();
            ctx.arc(centerX, centerY, radius, 0, 2 * Math.PI);  

            // to draw the line 
            const startX = 0;
            const startY = canvas1.height / 2;
            const endX = canvas1.width;
            const endY = canvas1.height / 2;
            ctx1.strokeStyle = "green"; // to assign the color
            ctx1.lineWidth = 3; // to assign the line thickness
            ctx1.beginPath();
            ctx1.moveTo(startX, startY); // Move cursor to start point
            ctx1.lineTo(endX, endY);  // Draw a line to the end point

            // to draw a point 
            const xPoint = canvas2.width / 2; // X coordinate of the point
            const yPoint = canvas2.height / 2;  // Y coordinate of the point
            ctx2.fillStyle = "blue"; //to assign the color
            ctx2.beginPath();
            ctx2.arc(xPoint, yPoint, 2, 0, 2 * Math.PI); // to draw a very small circle 
            ctx2.fill(); //Fill the circle with the specified color

             //to draw a rectangle
            const x =0; //  Top left corner x coordinate
            const y = 0;  // Top left corner y coordinate
            const width = 200; 
            const height = 100;
            ctx3.strokeStyle = "red";
            ctx3.lineWidth = 3; // to assign the line thickness
            ctx3.beginPath();
            ctx3.strokeRect(x, y, width, height); // to draw rectagle 
            
            // to make the outline drawing
            ctx.stroke();
            ctx1.stroke();
            ctx2.stroke();
            ctx3.stroke();
        </script>
